# Basic Controls

# Input

Image:

Define:

{
    "id": "text",
    "type": "input",
    "label": "Text",
    "default": "Lorem Ipsum is simply dummy text"
}

Advanced Attributes:

{
    "readonly": true // Định nghĩa input chỉ đọc
}

# Input Number

Image:

Define:

{
    "id": "test",
    "type": "input:number",
    "label": "Input Number TEST"
}

Advanced Attributes:

{
    "min": 10,
    "max": 100,
    "placeholder": "placeholder...",
    "readonly": true // Định nghĩa input chỉ đọc
}

# Input Unit

Image:

Define:

{
    "id": "test",
    "type": "input:unit",
    "label": "Input Unit TEST"
}

Advanced Attributes:

{
    "min": 10,
    "max": 100,
    "placeholder": "placeholder...",
    "units": ["em", "px"],
    "readonly": true // Định nghĩa input chỉ đọc
}

# Input Fix Content

Image:

Define:

{
    "id": "productSaleOff",
    "type": "input:fixContent",
    "label": "Product Sale Off",
    "default": "Sale off [!Profit!] %"
}

Advanced Attributes:

{
    "readonly": true // Định nghĩa input chỉ đọc
}

# Select

Image:

Define:

{
    "id": "tag",
    "type": "select",
    "label": "Tag",
    "default": "p",
    "options": [
        {
            "value": "p",
            "label": "P"
        },
        {
            "value": "span",
            "label": "Span"
        }
    ],
}
